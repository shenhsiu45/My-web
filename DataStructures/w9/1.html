Add Node in the Front of a Linked List 程式碼：

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next; 
};

struct node* addnode(struct node* start, int num) {
    struct node *x = (struct node*)malloc(sizeof(struct node));
    x ->data = num;
    x ->next = start;
    start = x;
    return start;
}

void printNode(struct node* start) {
    while (start != NULL) {
        printf("%d-->", start->data);
        start = start->next;
    }
    printf("NULL\n");
}

int length(struct node* start) {
    struct node* p = start;
    int count = 0;
    while (p != NULL) {
        count++;
        p = p->next;
    }
    return count;
}

int main() {
    struct node *head = NULL; 
    int n;
    
    while (scanf("%d", &n) != EOF) {
        head = addnode(head, n);
    }
    printNode(head);
    printf("The number of nodes: %d\n", length(head));
    
    return 0;
}
