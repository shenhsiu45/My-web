Add Node in the Tail of a Linked List (Version 2) 程式碼：

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next; 
};

void addNodeTail(struct node **start, struct node **tail, int num) {
    struct node *x = (struct node*)malloc(sizeof(struct node));
    x ->data = num;
    x ->next = NULL;
    if (*start == NULL && *tail == NULL) {
        *start = x;
        *tail = x;
    } else {
        (*tail)->next = x;
        *tail = x;
    }
}

void printNode(struct node* start) {
    while (start != NULL) {
        printf("%d-->", start->data);
        start = start->next;
    }
    printf("NULL\n");
}

int main() {
    struct node *head = NULL; 
    struct node *end = NULL; 
    int n;
    
    while (scanf("%d", &n) != EOF) {
        addNodeTail(&head, &end, n);
    }
    printNode(head);
    
    return 0;
}
