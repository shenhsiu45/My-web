Add Node in the Front of a Linked List (Version 2) 程式碼：

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next; 
};

void addNodeFront(struct node** start, int num) {
    struct node *x = (struct node*)malloc(sizeof(struct node));
    x ->data = num;
    x ->next = *start;
    *start = x;
}

void printNode(struct node* start) {
    while (start != NULL) {
        printf("%d-->", start->data);
        start = start->next;
    }
    printf("NULL\n");
}

int main() {
    struct node *head = NULL; 
    int n;
    
    while (scanf("%d", &n) != EOF) {
        addNodeFront(&head, n);
    }
    printNode(head);
    
    return 0;
}
