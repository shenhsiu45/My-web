Delete Front Node 程式碼：

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next; 
};

struct node* addnode(struct node* start, int num) {
    struct node *x = (struct node*)malloc(sizeof(struct node));
    x ->data = num;
    x ->next = start;
    start = x;
    return start;
}

void printNode(struct node* start) {
    while (start != NULL) {
        printf("%d-->", start->data);
        start = start->next;
    }
    printf("NULL\n");
}

// Delete front node
struct node* delnode(struct node* start) {
    struct node *p = start;
    if (start == NULL) {
        printf("The list is empty!\n");
        return NULL;
    }
    start = start->next;
    free(p);
    return start;
}

int main() {
    struct node *head = NULL; 
    int n;
    
    while (scanf("%d", &n) != EOF) {
        head = addnode(head, n);
    }
    printNode(head);
    
    while (head != NULL) {
        head = delnode(head);
        printNode(head);
    }
    
    return 0;
}
