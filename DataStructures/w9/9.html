Delete Tail Node

程式碼：

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next; 
};

struct node* addnode(struct node* start, int num) {
    struct node *x = (struct node*)malloc(sizeof(struct node));
    x ->data = num;
    x ->next = start;
    start = x;
    return start;
}

void printNode(struct node* start) {
    while (start != NULL) {
        printf("%d-->", start->data);
        start = start->next;
    }
    printf("NULL\n");
}

// Delete tail node
struct node* delnode2(struct node* start) {
    struct node *prev, *p;
    if (start == NULL) { // Empty list
        printf("The list is empty!\n");
        return NULL;
    } else if (start->next == NULL) { // Only one node
        p = start;
        free(p);
        return NULL;
    } else {
        prev = start;
        p = start->next;
        while (p->next != NULL) {
            prev = p;
            p = p->next;
        }
        prev->next = NULL;
        free(p);
        return start;
    }
}

int main() {
    struct node *head = NULL; 
    int n;
    
    while (scanf("%d", &n) != EOF) {
        head = addnode(head, n);
    }
    printNode(head);
    
    while (head != NULL) {
        head = delnode2(head);
        printNode(head);
    }

    return 0;
}
