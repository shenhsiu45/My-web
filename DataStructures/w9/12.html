#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next; 
};

struct node* addnode(struct node* start, int num) {
    struct node *x = (struct node*)malloc(sizeof(struct node));
    if (x != NULL) {
        x ->data = num;
        x ->next = start;
        start = x;
        return start;
    } else {
        return NULL;
    }
}

struct node* delnode(struct node* start) {
    struct node *temp = start;
    if (start == NULL) {
        printf("The list is empty!\n");
        return NULL;
    } else {
        printf("%d is pop!\n", start->data);
        start = start->next;
        free(temp);
        return start;
    }
}

void printNode(struct node* start) {
    while (start != NULL) {
        printf("%d-->", start->data);
        start = start->next;
    }
    printf("NULL\n");
}

int length(struct node* start) {
    int count = 0;
    while (start != NULL) {
        count++;
        start = start->next;
    }
    return count;
}

int main() {
    struct node *head = NULL; 
    int n;
    int i;
    int count;
    
    while (scanf("%d", &n) != EOF) {
        head = addnode(head, n);
        printNode(head);
    }
    
    count = length(head);
    printf("The number of nodes: %d\n", count);
    
    for (i = count; i > 0; i--) {
        head = delnode(head);
        printNode(head);
    }
    
    return 0;
}
