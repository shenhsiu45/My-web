Add Node in the Tail of a Linked List 程式碼：

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next; 
};

struct node* addnode2(struct node* start, int num) {
    struct node *p;
    struct node *x = (struct node*)malloc(sizeof(struct node));
    x ->data = num;
    x ->next = NULL;
    if (start == NULL) {
        start = x;
    } else {
        p = start;
        while (p ->next != NULL) {
            p = p ->next;
        }
        p ->next = x;
    }
    return start;
}

void printNode(struct node* start) {
    while (start != NULL) {
        printf("%d-->", start->data);
        start = start->next;
    }
    printf("NULL\n");
}

int main() {
    struct node *head = NULL; 
    int n;
    
    while (scanf("%d", &n) != EOF) {
        head = addnode2(head, n);
    }
    printNode(head);
    
    return 0;
}
