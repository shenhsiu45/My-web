Insert a Node in a Linked List (Version 2) 程式碼：

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next; 
};

void addnode3(struct node **start, int num) {
    struct node *prev, *current;
    struct node *ptr = (struct node*)malloc(sizeof(struct node));
    ptr ->data = num;
    ptr ->next = NULL;
    if (*start == NULL) {
        *start = ptr;
    } else {
        prev = NULL;
        current = *start;
        while (current != NULL && current->data > ptr->data) {
            prev = current;
            current = current->next;
        }
        if (prev == NULL) { //insert in the beginning of the list
            ptr->next = *start;
            *start = ptr;
        } else {
            ptr->next = current;
            prev->next = ptr;
        }
    }
}

void printNode(struct node* start) {
    while (start != NULL) {
        printf("%d-->", start->data);
        start = start->next;
    }
    printf("NULL\n");
}

int main() {
    struct node *head = NULL; 
    int n;
    
    while (scanf("%d", &n) != EOF) {
        addnode3(&head, n);
    }
    printNode(head);
    
    return 0;
}
