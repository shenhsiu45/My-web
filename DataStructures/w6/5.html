#include <stdio.h>

int linearSearch(int s[], int key, int n){
    int i;
    for(i=0; i<n; i++){
        if(s[i]==key){
            return i;
        }
    }
    return -1;
}

int main(){
    int s[10] = {10, 22, 45, 66, 77, 89, 99, 101, 120, 144};
    int key, pos;
    
    scanf("%d", &key);
    pos = linearSearch(s, key, 10);
    
    if(pos==-1){
        printf("%d cannot be found in this array!\n", key);
    }
    else{
        printf("%d is found at %d\n", key, pos);
    }
    
    return 0;
}
