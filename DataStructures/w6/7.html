Example of Binary Search Function程式碼：

#include <stdio.h>

int binarySearch(int s[], int key, int n){
    int lower = 0;
    int upper = n - 1;
    int mid;
    
    mid = (lower + upper) / 2;
    
    while(lower <= upper){
        if(s[mid] == key){
            return mid;
        }
        else{
            if(key > s[mid]){
                lower = mid + 1;
            }
            else{
                upper = mid - 1;
            }
        }
        mid = (lower + upper) / 2;
    }
    
    return -1;
}

int main(){
    int s[10] = {10, 22, 45, 66, 77, 89, 99, 101, 120, 144};
    int key, pos;
    
    scanf("%d", &key);
    
    pos = binarySearch(s, key, 10);
    
    if(pos == -1){
        printf("%d cannot be found in this array!\n", key);
    }
    else{
        printf("%d is found at %d\n", key, pos);
    }
    
    return 0;
}
