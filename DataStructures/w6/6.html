#include <stdio.h>

int main(){
    int s[10] = {10, 22, 45, 66, 77, 89, 99, 101, 120, 144};
    int key, i;
    int lower = 0;
    int upper = 9;
    int mid;
    
    scanf("%d", &key);
    
    mid = (lower + upper) / 2;
    while(lower <= upper){
        if(s[mid] == key){
            printf("%d is found at %d\n", key, mid);
            break;
        }
        else{
            if(key > s[mid]){
                lower = mid + 1;
            }
            else{
                upper = mid - 1;
            }
        }
        mid = (lower + upper) / 2;
    }
    
    if(lower > upper){
        printf("%d cannot be found in this array!\n", key);
    }
    
    return 0;
}
