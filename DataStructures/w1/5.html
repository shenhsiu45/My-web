#include <stdio.h>

int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int size = sizeof(arr) / sizeof(arr[0]);
    
    // 使用指標遍歷陣列並輸出元素
    printf("陣列元素：");
    for (int i = 0; i < size; i++) {
        printf("%d ", *(arr + i));
    }
    printf("\n");
    
    // 使用指標修改陣列元素
    int *ptr = arr; // 將指標指向陣列的第一個元素
    *(ptr + 2) = 75; // 修改第三個元素的值
    
    // 再次輸出修改後的陣列元素
    printf("修改後的陣列元素：");
    for (int i = 0; i < size; i++) {
        printf("%d ", *(arr + i));
    }
    printf("\n");
    
    return 0;
}
