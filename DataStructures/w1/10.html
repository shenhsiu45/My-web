#include <stdio.h>
#define SIZE 100

int main(){
    char s1[SIZE], s2[SIZE];
    int n1[SIZE]={0};
    int n2[SIZE]={0};
    int sum[SIZE]={0};
    int i, j;
    int carry = 0;
    scanf("%s", s1);
    scanf("%s", s2);
    for(i=0, j=strlen(s1)-1;j>=0;i++,j--){
        n1[i] = s1[j] - '0';
    }
    for(i=0, j=strlen(s2)-1;j>=0;i++,j--){
        n2[i] = s2[j] - '0';
    }
    carry = 0;
    for(i=0; i<SIZE; i++){
        sum[i] = n1[i] + n2[i] + carry;
        if(sum[i]>=10){
            carry = 1;
            sum[i] = sum[i] % 10;
        }else{
            carry = 0;
        }
    }
    for(i=SIZE-1; i>=0; i--){
        if(sum[i]!=0) break;
    }
    for(;i>=0; i--){
        printf("%d", sum[i]);
    }
    printf("\n");
    return 0;
}
