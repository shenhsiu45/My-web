d016: Postfix Notation 程式碼：

#include <stdio.h>
#include <string.h>

int stack[1000];
int top = -1;

void push(int);
int pop();

int main(){
    char equ[1000];
    char *tokenPtr;
    int m, n;
    
    while(gets(equ) != NULL){
        top = -1;
        tokenPtr = strtok(equ, " ");
        while ( tokenPtr != NULL ) {
            if(isdigit(*tokenPtr)){
                push(atoi(tokenPtr));
            }
            else{
                n = pop();
                m = pop();
                switch(*tokenPtr){
                    case '+':
                        push(m + n);
                        break;
                    case '-':
                        push(m - n);
                        break;
                    case '*':
                        push(m * n);
                        break;
                    case '/':
                        push(m / n);
                        break;
                    case '%':
                        push(m % n);
                        break;
                }   
            }   
            tokenPtr = strtok( NULL, " " );
        }
        printf("%d\n", pop());
    }
    
    return 0;
}

void push(int n){
    top++;
    stack[top] = n;
}

int pop(){
    int data = stack[top];
    top--;
    return data;
}
