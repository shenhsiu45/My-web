#include <stdio.h>
#define MAX 8

int main(){
    int s[MAX] = {10, 20, 30, 40, 50, 60, 70, 80};
    int key;
    int count = 0;
    int L = 0;
    int R = MAX-1;
    int M;
    scanf("%d", &key);
    while(L<=R){
        M = (L+R)/2;
        count++;
        if(s[M]==key){
            printf("%d in s[%d]\n", key, M);
            break;
        }
        else if(key > s[M]){
            L = M+1;
        }
        else{
            R = M-1;
        }
    }
    if(L>R){
        printf("%d is not in array!\n", key);
    }
    printf("count:%d\n", count);
    return 0;
}
