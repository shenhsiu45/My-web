a024. 最大公因數(GCD)

This program calculates the greatest common divisor (GCD) of two integers.

#include <stdio.h>

int GCD(int, int);

int main(){
    int a, b;
    while(scanf("%d%d", &a, &b)!=EOF){
        printf("%d\n", GCD(a, b));
    }
    return 0;
}

int GCD(int a, int b){
    int r;
    r = a % b;
    while(r!=0){
        a = b;
        b = r;
        r = a % b;
    }
    return b;
}
