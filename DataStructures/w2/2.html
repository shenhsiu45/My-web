function重要觀念: Call by value

呼叫者複製一份參數傳給function，function內部對該變數進行操作，不會改變原有呼叫者的參數值。


#include <stdio.h>

void add3(int, int, int);

int main(){
    int x, y, z;
    while(scanf("%d%d%d", &x, &y, &z) != EOF){
        printf("Before add3: %d %d %d\n", x, y, z);
        add3(x, y, z);
        printf("Outside add3: %d %d %d\n", x, y, z);
    }
    return 0;
}

void add3(int a, int b, int c){
    a += 3;
    b += 3;
    c = c + 3;
    printf("inside add3:%d %d %d\n", a, b, c);
}
