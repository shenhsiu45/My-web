Example: GCD

This program calculates the greatest common divisor (GCD) of two integers.

#include <stdio.h>

int main(){
    int a, b, r;
    while(scanf("%d%d", &a, &b)!=EOF){
        r = a % b;
        while(r!=0){
            a = b;
            b = r;
            r = a % b;
        }
        printf("%d\n", b);
    }
    return 0;
}
