Circular Queue Optimization

Below is the optimized version of the circular queue program:

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next;
};

void addCircularNode(struct node **start, struct node **tail, int num) {
    struct node *x = (struct node*)malloc(sizeof(struct node));

    if (*start == NULL) {
        x->data = num;
        x->next = x;
        *start = x;
        *tail = x;
    } else {
        x->data = num;
        x->next = *start;
        *start = x;
        (*tail)->next = *start;
    }
}

void printCircularNode(struct node* start) {
    struct node *p = start;
    while (p->next != start) {
        printf("%d-->", p->data);
        p = p->next;
    }
    printf("%d-->%d\n", p->data, p->next->data);
}

int main() {
    struct node *head = NULL;
    struct node *tail = NULL;
    int n;

    printf("Enter elements for the circular queue (enter any non-numeric value to end):\n");
    while (scanf("%d", &n) != EOF) {
        addCircularNode(&head, &tail, n);
    }
    printCircularNode(head);

    return 0;
}
