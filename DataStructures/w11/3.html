Add Node at Tail of Circular List

Below is the code to add a node at the tail of a circular list:

#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next;
};

struct node* addCircularNode(struct node* start, int num) {
    struct node *x = (struct node*)malloc(sizeof(struct node));
    struct node *p = start;
    if (start == NULL) {
        x->data = num;
        x->next = x;
        start = x;
    } else {
        while (p->next != start) {
            p = p->next;
        }
        x->data = num;
        x->next = start;
        p->next = x;
    }
    return start;
}

void printCircularNode(struct node* start) {
    struct node *p = start;
    while (p->next != start) {
        printf("%d-->", p->data);
        p = p->next;
    }
    printf("%d-->%d\n", p->data, p->next->data);
}

int main() {
    struct node *head = NULL;
    int n;

    printf("Enter elements for the circular list (enter any non-numeric value to end):\n");
    while (scanf("%d", &n) != EOF) {
        head = addCircularNode(head, n);
    }
    printCircularNode(head);

    return 0;
}
