e357: 遞迴函數練習

下面是計算 e357 函數的遞迴範例：

#include <stdio.h>

int F(int);

int main() {
    int n;
    printf("請輸入一個數字：");
    while (scanf("%d", &n) != EOF) {
        printf("%d\n", F(n));
    }
    return 0;
}

int F(int x) {
    if (x == 1) {
        return 1;
    } else if (x % 2 == 0) {
        return F(x / 2);
    } else {
        return F(x - 1) + F(x + 1);
    }
}
